Image Color Summarizer Example

Image Color Summarizer can analyse images and provide information about the colours contained within the image. For an explanation, see here. Unfortunately, it is not CORS enabled, so we need to use JSONP. It also does not allow us to specify a callback function, it automatically assumes we have a function named 'colorsummary' which can deal with the data that is returned: 

                
                    // Function that will process the response from the API
                    // This function should probably check for errors
                    var colorsummary = function(data) {
                        console.log(data);
                    }

                    // Function to ensure parameters used in request are encoded correctly
                    var encodeParameters = function(params) {
                        // join all key value pairs and store in an array
                        var strArray = [];
                        for(var key in params) {
                            if(params.hasOwnProperty(key)) {
                                var paramString = encodeURIComponent(key) + "=" + encodeURIComponent(params[key]);
                                strArray.push(paramString);
                            }
                        }
                        // join everything in the array together
                        return strArray.join("&");
                    }

                    // the parameters for the API request
                    var parameters = {
                        url: 'scontent.xx.fbcdn.net/t31.0-8/s720x720/12304416_10156352940115253_4201810718041006173_o.jpg',
                        precision: 'vlow',
                        num_clusters: 2,
                        jsonp: 1
                    }

                    // do the XHR request
                    var base_url = 'http://mkweb.bcgsc.ca/color-summarizer/';
                    var query_url = base_url + '?' + encodeParameters(parameters)
                    console.log(query_url);

                    // add a new script to the page, using the query_url as the source
                    var script = document.createElement('script');
                    script.src = query_url;
                    document.getElementsByTagName('head')[0].appendChild(script);
